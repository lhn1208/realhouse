				<script language="javascript" type="text/javascript">
					//<![CDATA[
					jQuery(function(){
						sTab();// 탭 호출
						fn_nav()// 페이지 네비 호출
					});
					function sTab(){
						var btn=jQuery('.s-tab-title a'),
						on='is-current';
						btn.on('click',function(e){
						if(!jQuery(this).hasClass(on)){
							jQuery(this).addClass(on).siblings('a').removeClass(on);
							jQuery(this).closest('.tab-section').find('.s-tab').css('display','none').eq(jQuery(this).index()).css('display','block');
						}
						e.preventDefault();
						}).eq(0).addClass(on);
					}
					function fn_nav(){
						fn_scroll();// 스크롤 호출
						jQuery(window).scroll(function(){
							fn_scroll();
						});
						function fn_scroll(){
							var winTop=jQuery(window).scrollTop(),
							s=jQuery('.s-1'),
							tit=s.find('.page-nav'),
							on='is-current';
							if(s.offset().top<winTop){
								tit.addClass('fixed');
							}else{
								tit.removeClass('fixed').find('a').removeClass('on');
							}
							jQuery('.mode-travel .section').each(function(i,el){
								$el=jQuery(el);
								if(938<winTop){
									if($el.offset().top-(tit.outerHeight()+1)<winTop){
										tit.find('a').removeClass(on).eq(i).addClass(on);
									}
								}
							});
						}
						jQuery('.mode-travel .page-nav a').on('click',function(e){
							var s=jQuery('.mode-travel .section'),
							i=jQuery(this).index();
							if(jQuery(this).hasClass('m-1')){
								jQuery('html,body').stop().animate({'scrollTop':s.eq(i).offset().top},'slow');
							}else if(jQuery(this).hasClass('m-6')){
								jQuery('html,body').stop().animate({'scrollTop':jQuery('.promotion_articles').offset().top-jQuery('.s-1 .page-nav').outerHeight()},'slow');
							}else{
								jQuery('html,body').stop().animate({'scrollTop':s.eq(i).offset().top-jQuery('.s-1 .page-nav').outerHeight()},'slow');
							}
							e.preventDefault();
						});
					}
					//]]>
				</script>
